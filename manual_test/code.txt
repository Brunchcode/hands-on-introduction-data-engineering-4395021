import pandas as pd
from datetime import date
df=pd.read_csv('data.csv')
df_agg_sp = df['Sector'].value_counts().reset_index()
df_agg_sp.columns = ['Sector','Count']
today = date.today()
df_agg_sp['Date']=today.strftime('%Y-%m-%d')
